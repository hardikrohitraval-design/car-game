import random def play_game(): print("🌟 Welcome to Number Guessing Adventure! 🌟") print("I'm thinking of a number between 1 and 100.") print("You have 8 guesses. Collect power-ups for clues!\n") secret_number = random.randint(1, 100) guesses_left = 8 power_ups = 2 # You start with 2 power-ups score = 0 while guesses_left > 0: try: guess = int(input(f"Guess the number (guesses left: {guesses_left}) → ")) except ValueError: print("Please enter a valid number!") continue guesses_left -= 1 score += 10 # Points for every guess attempt if guess == secret_number: print(f"\n🎉 AMAZING! You guessed it! The number was {secret_number}") print(f"Final Score: {score + (guesses_left * 20)} points! 🏆") break elif guess < secret_number: print("Too low! 📉") else: print("Too high! 📈") # Power-up system if guesses_left <= 4 and power_ups > 0 and random.random() > 0.5: power_ups -= 1 hint = "even" if secret_number % 2 == 0 else "odd" closeness = "very close" if abs(guess - secret_number) <= 5 else "somewhat close" print(f"✨ POWER-UP ACTIVATED! The number is {hint} and you're {closeness}!") if guesses_left == 0: print(f"\n💥 Game Over! The number was {secret_number}") print(f"Final Score: {score} points") # Play again? again = input("\nPlay again? (y/n): ").lower() if again == 'y': play_game() else: print("Thanks for playing! 👋") # Start the game if __name__ == "__main__": play_game()
